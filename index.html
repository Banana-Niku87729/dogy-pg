シンプルなコード例

// Dogyでのシンプルなプログラム例

// ウィンドウ表示
dogy window 600x500 "Sample Window" id=sample
// sampleと指定されたウィンドウにボタンを表示
dogy botton text="ボタン" id=sample y=100 px=14 btid=button1 color="#FF9999"
// クリック処理
clicked btid=button1 click_event
// 発火時に新しいウィンドウを表示。
click_event{dogy window 600x400 id=second "新しいウィンドウ"}
